마운트전에 컴포넌트를 생성해야 함

let vm = Vue.createApp({  })
vm.component('hello', {
  data() {
    return {
      message: "Hello World!"
    }
  }
  template: `<h1>{{ message }}</h1>`
});
vm.mount("#app");
